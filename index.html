const int drukknop = 2;
    const int schakeling = 7;

    int waarde = 0;

    bool state = false;
    void setup() {
        pinMode(schakeling, OUTPUT);
        pinMode(drukknop, INPUT);
    }

    void loop() {
        if(digitalRead(drukknop)){
            while(digitalRead(drukknop)==HIGH){
                delay(1);
            }
            switch(state){
                case true:
                    state = false;
                    digitalWrite(schakeling, LOW);
                    break;
                case false:
                    state = true;
                    digitalWrite(schakeling, HIGH);
                break;
            }
        }
    }
